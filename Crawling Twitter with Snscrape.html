3. Proses Crawling#

# membuat list sebagai output hasil crawling nantinya
tweets_list2 = []

# gunakan TwitterSearchScraper to mencari data and masukkan ke dalam variabel list yang sudah dibuat tadi
for i,tweet in enumerate(sntwitter.TwitterSearchScraper('Cak Imin since:2022-12-01 until:2023-02-21').get_items()):
    if i>500:
        break
    tweets_list2.append([tweet.date, tweet.id, tweet.content, tweet.user.username])
    
# membuat sebuah dataframe dari hasil tweets list dengan parameter dibawah ini
data = pd.DataFrame(tweets_list2, columns=['Datetime', 'Tweet Id', 'Text', 'Username'])
